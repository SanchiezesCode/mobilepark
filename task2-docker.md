Docker
Задача: необходимо дать рекомендации/правки по улучшению/исправлению следующего Docker file  # Используем базовый образ

 
FROM python
 
# Указываем рабочую директорию
WORKDIR /app
 
# Копируем файлы проекта
COPY . .
 
# Устанавливаем зависимости
RUN pip install -r requirements.txt
 
# Экспонируем порт для Flask
EXPOSE 5000
 
# Указываем команду запуска приложения
CMD ["python", "app.py"]`
